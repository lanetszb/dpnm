\end{gathered}
\end{cases}

\begin{gathered}
q^{m} = q^{m} (\bar{P_n}, t),

\begin{gathered}
\bar{P_n} = \frac{P_i+P_j}{2},

\begin{gathered}
q^m = q^{m'}(\bar{P}^{n+1}-\bar{P}^n+q^m(\bar{P}^n)),

\end{eqnarray}

\begin{eqnarray}
\begin{gathered}
q^{m'} = \frac{q^m(\bar{P}^n+\frac{\Delta{\tilde{P}}}{2})-q^m(\bar{P}^n-\frac{\Delta{\tilde{P}}}{2})}{\Delta{\tilde{P}}}
\end{gathered}
\end{eqnarray}
